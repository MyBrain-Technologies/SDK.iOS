void MBT_Fourier::forwardBluesteinFFT(std::vector<std::complex<float> >& inputData)
{
    forwardBluesteinFFT(inputData, Default);
}


void MBT_Fourier::forwardBluesteinFFT(std::vector<std::complex<float> >& inputData, const MBT_FourierOptions option)
{
    MBT_FourierBluestein::bluestein(inputData, getExponentSign(option));
    forwardScaling(inputData, option);
}


void MBT_Fourier::inverseBluesteinFFT(std::vector<std::complex<float> >& inputData)
{
    inverseBluesteinFFT(inputData, Default);
}


void MBT_Fourier::inverseBluesteinFFT(std::vector<std::complex<float> >& inputData, const MBT_FourierOptions option)
{
    MBT_FourierBluestein::bluestein(inputData, -getExponentSign(option));
    inverseScaling(inputData, option);
}


std::vector<std::complex<float> > MBT_Fourier::naiveForwardFFT(std::vector<std::complex<float> >& timeSpaceData, const MBT_FourierOptions option)
{
    std::vector<std::complex<float> > frequencySpaceData = naive(timeSpaceData, getExponentSign(option));
    forwardScaling(frequencySpaceData, option);
    return frequencySpaceData;
}


std::vector<std::complex<float> > MBT_Fourier::naiveInverseFFT(std::vector<std::complex<float> >& frequencySpaceData, const MBT_FourierOptions option)
{
    std::vector<std::complex<float> > timeSpaceData = naive(frequencySpaceData, -getExponentSign(option));
    inverseScaling(timeSpaceData, option);
    return timeSpaceData;
}


void MBT_Fourier::forwardRadix2FFT(std::vector<std::complex<float> >& inputData, const MBT_FourierOptions option)
{
    radix2(inputData, getExponentSign(option));
    forwardScaling(inputData, option);
}


void MBT_Fourier::inverseRadix2FFT(std::vector<std::complex<float> >& inputData, const MBT_FourierOptions option)
{
    radix2(inputData, -getExponentSign(option));
    inverseScaling(inputData, option);
}


std::vector<float> MBT_Fourier::frequencyValues(const int nbSample, const float sampRate)
{
    std::vector<float> scale;
    scale.resize(nbSample);

    float freq = 0;
    float step = sampRate / nbSample;
    int secondHalf = (nbSample >> 1) + 1;

    for (int i = 0; i < secondHalf; i++)
    {
        scale[i] = freq;
        freq += step;
    }

    freq = -step * (secondHalf - 2);
    for (int i = secondHalf; i < nbSample; i++)
    {
        scale[i] = freq;
        freq += step;
    }

    return scale;
}


int MBT_Fourier::getExponentSign(const MBT_FourierOptions option)
{
    return (option & InverseExponent) == InverseExponent ? 1 : -1;
}


void MBT_Fourier::forwardScaling(std::vector<std::complex<float> >& inputData, const MBT_FourierOptions option)
{
    if ((option & NoScaling) == NoScaling ||
        (option & AsymmetricScaling) == AsymmetricScaling)
    {
        return;
    }

    float scalingFactor = sqrt(1.0 / inputData.size());

    for (int i = 0; i < inputData.size(); i++)
    {
        inputData[i] *= scalingFactor;
    }
}


void MBT_Fourier::inverseScaling(std::vector<std::complex<float> >& inputData, const MBT_FourierOptions option)
{
    if ((option & NoScaling) == NoScaling)
    {
        return;
    }

    float scalingFactor = 1.0 / inputData.size();

    if ((option & AsymmetricScaling) != AsymmetricScaling)
    {
        scalingFactor = sqrt(scalingFactor);
    }

    for (int i = 0; i < inputData.size(); i++)
    {
        inputData[i] *= scalingFactor;
    }
}

template<typename T>
void MBT_Fourier::radix2Reorder(std::vector<T>& inputData)
{
    int j = 0;
    for (int i = 0; i < inputData.size() - 1; i++)
    {
        //Swap values
        if (i < j)
        {
            T temp = inputData[i];
            inputData[i] = inputData[j];
            inputData[j] = temp;
        }

        int len = (int)inputData.size();

        do
        {
            len >>= 1;
            j ^= len;
        }
        while ((j & len) == 0);
    }
}


void MBT_Fourier::radix2Step(std::vector<std::complex<float> >& inputData, const int exponentSign, const int levelSize, const int indexInLevel)
{
    // Twiddle Factor
    const float  M_PI=3.14159265358979f;
    float exponent = (exponentSign * indexInLevel) * M_PI / levelSize;
    std::complex<float> w (cos(exponent), sin(exponent));

    int step = levelSize << 1;
    for (int i = indexInLevel; i < inputData.size(); i += step)
    {
        std::complex<float> value = inputData[i];
        std::complex<float> modifier = w * inputData[i + levelSize];
        inputData[i] = value + modifier;
        inputData[i + levelSize] = value - modifier;
    }
}


void MBT_Fourier::radix2(std::vector<std::complex<float> >& inputData, const int exponentSign)
{
    int len = (int)inputData.size();
    if (len == 0 || (len & (len - 1))) //If the length of inputData is not a positive power of two.
    {
        errno = EINVAL;
        perror("ERROR: FFT CANNOT BE COMPUTED WITH RADIX2 WHEN THE DATA LENGTH IS NOT A POSITIVE POWER OF TWO");
    }

    radix2Reorder<std::complex<float> >(inputData);

    for (int levelSize = 1; levelSize < len; levelSize *= 2)
    {
        for (int indexInLevel = 0; indexInLevel < levelSize; indexInLevel++)
        {
            radix2Step(inputData, exponentSign, levelSize, indexInLevel);
        }
    }
}


std::vector<std::complex<float> > MBT_FourierBluestein::bluesteinSequence(const int nbSample)
{
    const float  M_PI=3.14159265358979f;
    float multiplier = M_PI / nbSample;
    std::vector<std::complex<float> > sequence;
    sequence.resize(nbSample);

    if (nbSample > sqrt(LONG_MAX)) //sqrt(LONG_MAX) is the maximum possible length for the Bluestein sequence before overflow.
    {
        for (int k = 0; k < nbSample; k++)
        {
            float t = (multiplier * k) * k; //Avoiding overflow.
            sequence[k] = std::complex<float>(cos(t), sin(t));
        }
    }
    // TODO: benchmark whether the second variation is significantly
    // faster than the former one. If not just use the former one always.
    else
    {
        for (int k = 0; k < nbSample; k++)
        {
            float t = multiplier * (k * k);
            sequence[k] = std::complex<float>(cos(t), sin(t));
        }
    }

    return sequence;
}


//Helper functions and structures for MBT_Fourier::bluesteinConvolutionParallel.
typedef struct {
    std::vector<std::complex<float> >* inputData;
    std::vector<std::complex<float> >* a;
    std::vector<std::complex<float> >* sequence;
    int len;
} WorkerArgBluesteinConvolutionParallelA;

typedef struct {
    int len;
    int powerOfTwoLength;
    std::vector<std::complex<float> >* b;
    std::vector<std::complex<float> >* sequence;
} WorkerArgBluesteinConvolutionParallelB;

void * MBT_FourierBluestein::workerBluesteinConvolutionParallelA(void * arg){
    int len = ((WorkerArgBluesteinConvolutionParallelA *) arg)->len;
    std::vector<std::complex<float> >* inputData= ((WorkerArgBluesteinConvolutionParallelA *) arg)->inputData;
    std::vector<std::complex<float> >* a= ((WorkerArgBluesteinConvolutionParallelA *) arg)->a;
    std::vector<std::complex<float> >* sequence = ((WorkerArgBluesteinConvolutionParallelA *) arg)->sequence;

    for (int i = 0; i < len; i++)
    {
        a->at(i) = std::conj(sequence->at(i)) * inputData->at(i);
    }

    MBT_Fourier::radix2(*a, -1);

    return NULL;
}

void * MBT_FourierBluestein::workerBluesteinConvolutionParallelB(void * arg){
    int len = ((WorkerArgBluesteinConvolutionParallelB *) arg)->len;
    int powerOfTwoLength = ((WorkerArgBluesteinConvolutionParallelB *) arg)->powerOfTwoLength;
    std::vector<std::complex<float> >* b = ((WorkerArgBluesteinConvolutionParallelB *) arg)->b;
    std::vector<std::complex<float> >* sequence = ((WorkerArgBluesteinConvolutionParallelB *) arg)->sequence;

    for (int i = 0; i < len; i++)
    {
        b->at(i) = sequence->at(i);
    }

    for (int i = powerOfTwoLength - len + 1; i < powerOfTwoLength; i++)
    {
        //b->at(i) = sequence->at(powerOfTwoLength - i);
        b->at(i) = 0; // Fanny Grosselin 2017/01/12
    }

    MBT_Fourier::radix2(*b, -1);

    return NULL;
}

void MBT_FourierBluestein::bluesteinConvolutionParallel(std::vector<std::complex<float> >& inputData)
{
    int len = (int)inputData.size();
    std::vector<std::complex<float> > sequence = bluesteinSequence(len);

    // Padding to power of two >= 2N–1 so we can apply Radix-2 FFT.
    //int powerOfTwoLength = pow(2, ceil(log(2 * len - 1)/log(2)));
    int powerOfTwoLength = pow(2, floor(log(2 * len - 1)/log(2))); // Fanny Grosselin 2017/01/19
    std::vector<std::complex<float> > b;
    b.assign(powerOfTwoLength, 0);
    std::vector<std::complex<float> > a;
    a.assign(powerOfTwoLength, 0);

    //Without multithreading
    // Build and transform padded sequence b_k = exp(I*Pi*k^2/N)
    for (int i = 0; i < len; i++)
    {
        b[i] = sequence[i];
    }

    for (int i = powerOfTwoLength - len + 1; i < powerOfTwoLength; i++)
    {
        //b[i] = sequence[powerOfTwoLength - i];
        b[i] = 0; // Fanny Grosselin 2017/01/12
    }

    MBT_Fourier::radix2(b, -1);

    // Build and transform padded sequence a_k = x_k * exp(-I*Pi*k^2/N)
    for (int i = 0; i < len; i++)
    {
        a[i] = std::conj(sequence[i]) * inputData[i];
    }

    MBT_Fourier::radix2(a, -1);

    for (int i = 0; i < a.size(); i++)
    {
        a[i] *= b[i];
    }

    MBT_Fourier::radix2(a, 1);

    /*

    float nbInv = 1.0 / powerOfTwoLength;

    for (int i = 0; i < len; i++)
    {
        inputData[i] = nbInv * std::conj(sequence[i]) * a[i];
    }

void MBT_FourierBluestein::swapComplexParts(std::vector<std::complex<float> >& inputData)
{
    for (int i = 0; i < inputData.size(); i++)
    {
        inputData[i] = std::complex<float>(inputData[i].imag(), inputData[i].real());
    }
}


void MBT_FourierBluestein::bluestein(std::vector<std::complex<float> >& inputData, const int exponentSign)
{
    int len = (int)inputData.size();
    if (len != 0 && !(len & (len - 1)))
    {
        //std::cerr << "RADIX2" << std::endl;
        MBT_Fourier::radix2(inputData, exponentSign);
        return;
    }

    if (exponentSign == 1)
    {
        swapComplexParts(inputData);
    }

    //std::cerr << "BLUESTEINCONVOLUTION" << std::endl;
    bluesteinConvolutionParallel(inputData);

    if (exponentSign == 1)
    {
        swapComplexParts(inputData);
    }
}
